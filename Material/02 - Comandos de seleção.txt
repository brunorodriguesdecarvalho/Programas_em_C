#include <stdio.h>

int main(void) {
   float a, b, c;
   printf("Numeros? ");
   scanf("%f %f %f",&a,&b,&c);
   if( a<b+c && b<a+c && c<a+b ) {
      printf("Triangulo: ");
      if( a==b && b==c ) puts("equilatero");
      else if( a==b || a==c || b==c) puts("isosceles");
      else puts("escaleno");
   }
   else puts("Nao e triangulo!");
   return 0;
}

int main(void) {
   int n;
   printf("Numero? ");

   switch( n ) {
      case  1: putchar('A'); break;
      case  3: putchar('B');
      case  4: putchar('C'); break;
      default: putchar('*');
      case  5: putchar('D');
   }
   puts("."); 
   return 0;
}

// Exemplo 10 
//------------------------------------------------------------

#include <stdio.h>

int main(void) {
   float x, y;
   char o;
   printf("Expressao? ");
   scanf("%f %c %f",&x,&o,&y);
   switch( o ) {
      case '+': printf("Valor = %.2f\n",x+y); break;
      case '-': printf("Valor = %.2f\n",x-y); break;
      case '*': printf("Valor = %.2f\n",x*y); break;
      case '/': printf("Valor = %.2f\n",x/y); break;
      default : printf("Operador invalido: %c\n",o);
   }
   return 0;
}
